# DjangoProject

## 가상환경 만들기 (터미널에 입력)
1. py -m venv venv
2. venv\Scripts\activate
3. py pip -m install django
4. py pip -m install python-steam-api
5. py pip -m install beautifulsoup4

---

## .env 파일 생성 후 api key 입력 (필수 아님)
- echo "STEAM_API_KEY=<YOUR_STEAM_API KEY>" >> .env
